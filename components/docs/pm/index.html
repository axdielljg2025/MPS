Package Manager

Source
0.8.0

Usage

The package managers shown can be configured in main.pm.

Install

npm
pnpm
bun
yarn
npm i shadcn-docs-nuxt

pnpm i shadcn-docs-nuxt

bun add shadcn-docs-nuxt

yarn add shadcn-docs-nuxt
npm
pnpm
bun
yarn
npm i -D shadcn-docs-nuxt

pnpm i -D shadcn-docs-nuxt

bun add -d shadcn-docs-nuxt

yarn add -D shadcn-docs-nuxt
npm
pnpm
bun
yarn
npm install

pnpm install

bun install

yarn install
npm
pnpm
bun
yarn
npm i -g nuxi@latest

pnpm i -g nuxi@latest

bun add -g nuxi@latest

yarn add -g nuxi@latest

:pm-install{name="shadcn-docs-nuxt"}
:pm-install{name="shadcn-docs-nuxt" save-dev}
:pm-install
:pm-install{name="-g nuxi@latest"}

Run

npm
pnpm
bun
yarn
npm run dev

pnpm run dev

bun run dev

yarn run dev
npm
pnpm
bun
yarn
npm run build --watch -o

pnpm run build --watch -o

bun run build --watch -o

yarn run build --watch -o

:pm-run{script="dev"}
:pm-run{script="build --watch -o"}

X

npm
pnpm
bun
yarn
npx nuxi@latest init <project-name>

pnpm dlx nuxi@latest init <project-name>

bun x nuxi@latest init <project-name>

yarn dlx nuxi@latest init <project-name>

:pm-x{command="nuxi@latest init <project-name>"}

No Sync

npm
pnpm
bun
yarn
npx nuxi@latest init <project-name>

pnpm dlx nuxi@latest init <project-name>

bun x nuxi@latest init <project-name>

yarn dlx nuxi@latest init <project-name>

:pm-x{command="nuxi@latest init <project-name>" no-sync}
